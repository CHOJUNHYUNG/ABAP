FORM get_ddtext .
  SELECT *
    FROM dd03l AS l3 LEFT OUTER JOIN dd07t AS t7 ON l3~domname = t7~domname
    INTO CORRESPONDING FIELDS OF TABLE gt_domain
    WHERE l3~tabname = gc_table
      AND l3~domname LIKE 'Z%'.

  SORT gt_domain BY fieldname domvalue_l.

  SELECT fieldname
    FROM dd03l INTO TABLE gt_fieldname
    WHERE tabname = gc_table
      AND domname LIKE 'Z%'.
ENDFORM.

*&---------------------------------------------------------------------*
FORM build_data .

  DATA lv_rc TYPE c VALUE 0.

  LOOP AT gt_excel INTO gs_excel.
    CLEAR gs_disp.
    gs_disp-icon = icon_led_yellow.
    gs_disp-name = gs_excel-name.
    gs_disp-waers = gc_default_cury.
    gs_disp-address = gs_excel-address.

    " ## ## ## (##, ##, ##, ##, ####, ##, ###)
    PERFORM check_required USING    gs_excel
                           CHANGING gs_disp  lv_rc.

    IF lv_rc <> 0.
      APPEND gs_disp TO gt_disp.
      CONTINUE.
    ENDIF.

    " ## ## ## (format: AI-A-xx, length: 7)
    PERFORM check_empid USING    gs_excel
                        CHANGING gs_disp.

    " ddtext
    PERFORM check_ddtext USING    gs_excel
                         CHANGING gs_disp.

    " ###, ###  ##
    PERFORM check_date  USING    gs_excel
                        CHANGING gs_disp.

    "##
    PERFORM check_salary USING    gs_excel
                         CHANGING gs_disp.

    APPEND gs_disp TO gt_disp.
  ENDLOOP.


ENDFORM.

FORM check_empid USING p_excel LIKE gs_excel
                  CHANGING p_disp  LIKE gs_disp.

  DATA lt_empid LIKE gt_empid.

  " empid (key) ## ##
  CLEAR gv_empid.
  READ TABLE gt_empid INTO gv_empid WITH KEY gs_excel-empid.
  IF sy-subrc = 0.
    gs_disp-icon = icon_led_red.

  ELSE.
    gv_empid = gs_excel-empid.
    APPEND gv_empid TO gt_empid.
  ENDIF.

  " empid format, ## ##
  IF strlen( gs_excel-empid ) = 7.
    IF gs_excel-empid+0(5) <> 'AI-A-'.
      gs_disp-icon = icon_led_red.
      gs_disp-message = '## ## ##'.
    ENDIF.
  ELSE.
    gs_disp-icon = icon_led_red.

*&---------------------------------------------------------------------*
*&      --> GS_EXCEL
*&      <-- GS_DISP
*&---------------------------------------------------------------------*
FORM check_date USING p_excel LIKE gs_excel
                 CHANGING p_disp  LIKE gs_disp.

  DATA: lv_date  TYPE char10,
        lv_inpdt TYPE sy-datum,
        lv_retdt TYPE sy-datum.

  CLEAR: lv_inpdt, lv_retdt.
  " ## ### ##.
  IF p_excel-inpdt IS NOT INITIAL.
    lv_date = p_excel-inpdt.
    REPLACE ALL OCCURRENCES OF '-' IN lv_date WITH ''.
    REPLACE ALL OCCURRENCES OF '/' IN lv_date WITH ''.
    REPLACE ALL OCCURRENCES OF '.' IN lv_date WITH ''.
    lv_inpdt = lv_date.
  ENDIF.

  IF p_excel-retdt IS NOT INITIAL.
    CLEAR lv_date.
    lv_date = p_excel-retdt.
    REPLACE ALL OCCURRENCES OF '-' IN lv_date WITH ''.
    REPLACE ALL OCCURRENCES OF '/' IN lv_date WITH ''.
    REPLACE ALL OCCURRENCES OF '.' IN lv_date WITH ''.
    lv_retdt = lv_date.
  ENDIF.

  p_disp-inpdt = lv_inpdt.
  p_disp-retdt = lv_retdt.

  " ## ## ##.
  CALL FUNCTION 'DATE_CHECK_PLAUSIBILITY'
    EXPORTING
      date                      = lv_inpdt
    EXCEPTIONS
      plausibility_check_failed = 1
      OTHERS                    = 2.
  IF sy-subrc <> 0.
    p_disp-icon = icon_led_red.
    p_disp-message = '### ## ##:yyyy.mm.dd'.
    EXIT.
  ENDIF.

  IF lv_retdt IS NOT INITIAL.
    CALL FUNCTION 'DATE_CHECK_PLAUSIBILITY'
      EXPORTING
        date                      = lv_retdt
      EXCEPTIONS
        plausibility_check_failed = 1
        OTHERS                    = 2.
    IF sy-subrc <> 0.
      p_disp-icon = icon_led_red.
      p_disp-message = '### ## ##:yyyy.mm.dd'.
      EXIT.
    ENDIF.
  ENDIF.

  " ##### ## ## ## ##.
  CASE p_excel-type.
    WHEN '1' OR '2'.
      IF lv_retdt IS NOT INITIAL.
        p_disp-icon = icon_led_red.
        p_disp-message = '#### ### ## ##'.
      ENDIF.
    WHEN '3'.
      IF lv_retdt IS INITIAL.
        p_disp-icon = icon_led_red.
        p_disp-message = '#### ### ## ##'.
      ELSE.
        IF lv_inpdt > lv_retdt.
          p_disp-icon = icon_led_red.
          p_disp-message = '### > ###'.
        ENDIF.
      ENDIF.
  ENDCASE.


ENDFORM.

*&---------------------------------------------------------------------*
*&      --> GS_EXCEL
*&      <-- GS_DISP
*&---------------------------------------------------------------------*
FORM check_ddtext USING p_excel
                   CHANGING p_disp.

  DATA: lv_fld_xl TYPE char20,
        lv_fld_dp TYPE char20.

  FIELD-SYMBOLS: <lf_xl>, <lf_dp>.

  LOOP AT gt_fieldname INTO gv_fieldname.
    lv_fld_xl = 'P_EXCEL-' && gv_fieldname.
    ASSIGN (lv_fld_xl) TO <lf_xl>.
    lv_fld_dp = 'P_DISP-' && gv_fieldname.
    ASSIGN (lv_fld_dp) TO <lf_dp>.
    <lf_dp> = <lf_xl>.

    READ TABLE gt_domain INTO gs_domain WITH KEY fieldname = gv_fieldname
                                                 domvalue_l = <lf_xl>.
    IF sy-subrc = 0.
      lv_fld_dp = 'P_DISP-' && gv_fieldname && '_T'.
      ASSIGN (lv_fld_dp) TO <lf_dp>.
      <lf_dp> = gs_domain-ddtext.
    ENDIF.
  ENDLOOP.

ENDFORM.

    EXPORTING

   IMPORTING

*     RETURN               =

FORM save_data .

  DATA: ls_data TYPE z24t0020,
        lt_data TYPE TABLE OF z24t0020.

  READ TABLE gt_disp INTO gs_disp WITH KEY icon = ICON_LED_RED.
  IF sy-subrc = 0.
    MESSAGE S000 WITH '## #### ### ### # #####.' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.

  CLEAR gs_disp.
  LOOP AT gt_disp INTO gs_disp.
    MOVE-CORRESPONDING gs_disp TO ls_data.
    ls_data-erdat = sy-datum.
    ls_data-erzet = sy-uzeit.
    ls_data-ernam = sy-uname.
    APPEND ls_data TO lt_data.
  ENDLOOP.

*  INSERT z24t0020 FROM TABLE lt_data ACCEPTING DUPLICATE KEYS.
  IF sy-subrc = 0.
    COMMIT WORK.
    CLEAR gs_disp.
    LOOP AT gt_disp INTO gs_disp.
      gs_disp-icon = ICON_LED_GREEN.
      CLEAR gs_disp-message.
      MODIFY gt_disp FROM gs_disp TRANSPORTING icon message.
      CLEAR gs_disp.
    ENDLOOP.
    MESSAGE S000 WITH '## ##'.
    gv_saved = 'X'.
  ELSE.
    ROLLBACK WORK.
    MESSAGE S000 WITH '## ##' DISPLAY LIKE 'E'.
    EXIT.
  ENDIF.


ENDFORM.
