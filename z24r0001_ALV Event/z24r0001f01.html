FORM get_data .

  SELECT *
    FROM mara LEFT OUTER JOIN makt ON mara~matnr = makt~matnr AND makt~spras = sy-langu
    INTO CORRESPONDING FIELDS OF TABLE gt_mara.

  SORT gt_mara BY matnr.

  SELECT *
    FROM z24t0020 INTO CORRESPONDING FIELDS OF TABLE gt_disp.

  LOOP AT gt_disp INTO gs_disp.
    gs_disp-icon = icon_led_green.
    MODIFY gt_disp FROM gs_disp.
  ENDLOOP.

  SORT gt_disp BY empid.
ENDFORM.

                              p_column_id TYPE LVC_S_COl.

  READ TABLE gt_mara INTO gs_mara INDEX p_row_id-index.
