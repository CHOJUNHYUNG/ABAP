MODULE set_detail_0200 OUTPUT.
  LOOP AT SCREEN.
    IF screen-name = 'GS_DISP_0200-EMPID'.
      CASE cp_ok_code.
        WHEN 'CREATE'.
          screen-required = '1'.
        WHEN 'CHANGE'.
          screen-input = '0'.
      ENDCASE.
      MODIFY SCREEN.
    ENDIF.

    IF screen-name = 'GS_DISP_0200-INPDT'.
      IF gs_disp_0200-type = '1'.
        screen-required = '1'.
        MODIFY SCREEN.
      ENDIF.
    ELSEIF screen-name = 'GS_DISP_0200-RETDT'.
      IF gs_disp_0200-type = '3'.
        screen-required = '1'.
        MODIFY SCREEN.
      ENDIF.
    ENDIF.
  ENDLOOP.
ENDMODULE.
