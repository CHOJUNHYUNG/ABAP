FORM HANDLE_NODE_DOUBLE_CLICK USING P_NODE_KEY TYPE TV_NODEKEY.

  CASE P_NODE_KEY.
    WHEN 'AA'.
      CALL TRANSACTION 'MD61'.
    WHEN 'AB'.
      CALL TRANSACTION 'MD02'.
*      SUBMIT ZAI01R01_01 VIA SELECTION-SCREEN WITH P_RA1 = '' WITH  P_RA2 = 'X' AND RETURN.
    WHEN 'AC'.
      CALL TRANSACTION 'MD04'.
    WHEN 'AD'.
      CALL TRANSACTION 'ZAI1PP01'.
    WHEN 'AE'.
      CALL TRANSACTION 'CO01'.
    WHEN 'BA'.
      CALL TRANSACTION 'ME57'.
    WHEN 'BB'.
      CALL TRANSACTION 'ME2ON'.
    WHEN 'BC'.
      CALL TRANSACTION 'ZAI1MM01'.
    WHEN 'BD'.
      CALL TRANSACTION 'ZAI1MM02'.
    WHEN 'CA'.
      CALL TRANSACTION 'ZAI1SD01'.
  ENDCASE.
ENDFORM.
