  MESSAGE S006 INTO GV_GRID_TITLE WITH SY-DBCNT.  "## # # : &

    GT_TTIC-OPERATION  = '0010'.                   "##### ##
    GT_TTIC-POSTG_DATE = SY-DATUM.                 "###### ####
    GT_TTIC-PLANT      = 1010.                     "### ##
    GT_TTIC-YIELD      = GS_DATA-GAMNG.            "####

    APPEND GT_TTIC.

    CLEAR GT_TTIC.
  ENDLOOP.

  CALL FUNCTION 'BAPI_PRODORDCONF_GET_TT_PROP'
    EXPORTING

    IMPORTING
      RETURN             = GS_RETURN
    TABLES
      TIMETICKETS        = GT_TTIC
      GOODSMOVEMENTS     = GT_GOODS
      LINK_CONF_GOODSMOV = GT_LINK
      DETAIL_RETURN      = GT_DETAIL.

  MOVE-CORRESPONDING GT_TTIC[] TO GT_DISP1.

  LOOP AT GT_DISP1 INTO GS_DISP1.
    READ TABLE GT_DATA INTO GS_DATA INDEX SY-TABIX.
    GS_DISP1-MATNR = GS_DATA-MATNR.
    GS_DISP1-MAKTX = GS_DATA-MAKTX.
    GS_DISP1-ICON  = ICON_SELECT_DETAIL.
    GS_DISP1-AEDAT = GS_DATA-AEDAT.

    GS_DISP1-AENAM = GS_DATA-AENAM.

      LS_BOM-INFO  = 'C600'.
