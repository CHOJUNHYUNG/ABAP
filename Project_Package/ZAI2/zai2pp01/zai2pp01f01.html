FORM GET_DATA .


  CLEAR : GS_DISP ,GT_DISP.

  DATA : LS_ROWS TYPE LVC_S_ROW,
         LT_ROWS TYPE LVC_T_ROW.



  SELECT DISTINCT A~*,B~MATNR,B~MAKTX,C~WERKS,C~ARBPL ,D~ERNAM ,D~ERDAT ,D~ERFZEIT,
                  E~RUECK, E~ERSDA, E~VORNR ,E~ERSDA, E~LMNGA, E~MEINH, KO~GMEIN,
                  KO~GAMNG ,G~VGW01, G~VGW02 , H~MBLNR, H~BWART, H~LGORT, H~BLDAT, H~BUDAT, "DISTINCT ### ##.
                  H~KZEAR, H~CHARG

    INTO CORRESPONDING FIELDS OF TABLE @GT_DISP
    FROM AFPO AS A LEFT JOIN  MAKT  AS B  ON A~MATNR  = B~MATNR
                   LEFT JOIN  AFKO  AS KO ON A~AUFNR  = KO~AUFNR
                   LEFT JOIN  AFVC  AS VC ON KO~AUFPL = VC~AUFPL
                   LEFT JOIN  CRHD  AS C  ON VC~ARBID = C~OBJID
                   LEFT JOIN  AFRU  AS E  ON A~AUFNR  = E~AUFNR
                   LEFT JOIN  AUFK  AS D  ON A~AUFNR  = D~AUFNR
                   LEFT JOIN  AUFM  AS H  ON D~AUFNR  = H~AUFNR
                   LEFT JOIN  AFVV  AS G  ON VC~AUFPL = G~AUFPL


     WHERE D~AUFNR IN @S_ORDER
      AND  B~MATNR IN @S_MAT
      AND  C~ARBPL IN @S_WORK
      AND  D~ERNAM IN @S_ERNAM
      AND  D~ERDAT IN @S_ERDAT
     "AND  D~ERNAM LIKE 'PAI-A%'
      AND  B~SPRAS = 'E'.        "E=EN D=DE
  "AND  H~BWART IN ( '101' , @SPACE ) .

  LOOP AT GT_DISP INTO GS_DISP.

    SELECT  KZEAR AUFNR
    FROM RESB
    INTO CORRESPONDING FIELDS OF TABLE GT_DISPS
    WHERE AUFNR = GS_DISP-AUFNR.


    READ TABLE GT_DISPS INTO GS_DISPS WITH KEY  AUFNR = GS_DISP-AUFNR.

    GS_DISP-KZEAR = GS_DISPS-KZEAR .
    MODIFY GT_DISP FROM GS_DISP.

  ENDLOOP.



  DELETE  GT_DISP  WHERE BWART NE '101' AND BWART NE SPACE .

  GS_DISP-ICON = ICON_CHECKED.
  MODIFY GT_DISP FROM GS_DISP TRANSPORTING ICON WHERE RUECK NE SPACE.

  GS_DISP-ICON3 = ICON_CHECKED.
  MODIFY GT_DISP FROM GS_DISP TRANSPORTING ICON3 WHERE MBLNR NE ''.

  GS_DISP-ICON5 = ICON_CHECKED.
  MODIFY GT_DISP FROM GS_DISP TRANSPORTING ICON5 WHERE KZEAR EQ 'X'.

  SORT GT_DISP BY AUFNR DESCENDING.
  LOOP AT GT_DISP INTO GS_DISP.
    APPEND GS_DISP TO GT_DISP2 .CLEAR GS_DISP.
  ENDLOOP.
  CLEAR GV_GRID_TITLE.
  MESSAGE S010 INTO GV_GRID_TITLE WITH CONV I( CONV STRING( LINES( GT_DISP2 ) ) ).

ENDFORM.
