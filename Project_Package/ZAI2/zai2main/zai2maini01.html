MODULE exit_0100 INPUT.
*  gv_ucomm = ok_code.
*  CLEAR : ok_Code.

  CASE ok_code.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'EXIT' OR 'CANC'.
      LEAVE PROGRAM.
    WHEN OTHERS.
  ENDCASE.
ENDMODULE.

MODULE user_command_0100 INPUT.
*  CLEAR gv_ucomm.
*  gv_ucomm = ok_code.
*  CLEAR ok_code.
  CASE ok_code.
    WHEN 'EDIT'.
      PERFORM board_edit.
    WHEN 'SAVE'.
      PERFORM board_save.
*    WHEN OTHERS.
  ENDCASE.

ENDMODULE.
